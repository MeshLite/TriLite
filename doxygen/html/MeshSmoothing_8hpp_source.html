24 // Implementation of VolumePreservingLaplacianSmooth
25 void VolumePreservingLaplacianSmooth(TL::Trimesh& mesh, int iterations, double lambda, double mu) {
26 std::vector<Eigen::Vector3d> new_positions(mesh.NumVertices());
27
28 for (int iter = 0; iter < iterations; ++iter) {
29 // Laplacian smoothing step (positive weight)
30 for (TL::Vertex v : mesh.Vertices()) {
31 Eigen::Vector3d laplacian(0, 0, 0);
32 size_t valence = 0;
33 for (TL::Halfedge h : mesh.VOutgoings(v)) {
34 laplacian += mesh.VPosition(mesh.HDestination(h));
35 ++valence;
36 }
37 laplacian /= static_cast<double>(valence);
38 new_positions[v] = mesh.VPosition(v) + lambda * (laplacian - mesh.VPosition(v));
39 }
40 for (TL::Vertex v : mesh.Vertices()) {
41 mesh.VPosition(v) = new_positions[v];
42 }
43
44 // Laplacian smoothing step (negative weight, inverse smoothing)
45 for (TL::Vertex v : mesh.Vertices()) {
46 Eigen::Vector3d laplacian(0, 0, 0);
47 size_t valence = 0;
48 for (TL::Halfedge h : mesh.VOutgoings(v)) {
49 laplacian += mesh.VPosition(mesh.HDestination(h));
50 ++valence;
51 }
52 laplacian /= static_cast<double>(valence);
53 new_positions[v] = mesh.VPosition(v) + mu * (laplacian - mesh.VPosition(v));
54 }
55 for (TL::Vertex v : mesh.Vertices()) {
56 mesh.VPosition(v) = new_positions[v];
57 }
58 }
59 }

20 void DecimateMesh(Trimesh& mesh, size_t target_face_count) {
21 auto compare = [&mesh](const auto& a, const auto& b) { return a.second > b.second; };
22 std::priority_queue<std::pair<Halfedge, double>, std::vector<std::pair<Halfedge, double>>, decltype(compare)> hqueue(compare);
23 for (Halfedge h : mesh.Halfedges()) { hqueue.push(std::make_pair(h, mesh.HLength(h))); }
24 while (mesh.NumFaces() > target_face_count && !hqueue.empty()) {
25 auto [minh, length] = hqueue.top(); hqueue.pop();
26 if(minh >= mesh.NumHalfedges() || mesh.HLength(minh) != length) { continue; }
27 std::array<Vertex, 2> verts = { mesh.HOrigin(minh), mesh.HDestination(minh) };
28 Vertex last_vert_id = mesh.NumVertices()-1;
29 auto [rm_faces, rm_vertices] = mesh.EdgeCollapse(minh);
30 for (const Vertex& rem_v : rm_vertices) {
31 for (Vertex& v : verts) {
32 if (v == rem_v) { v = kInvalidId; }
33 else if (v == last_vert_id) { v = rem_v; }
34 }
35 last_vert_id--;
36 }
37 for(Face f : rm_faces) {
38 for(Halfedge h : {3*f, 3*f+1, 3*f+2}) if(h < mesh.NumHalfedges()) { hqueue.push(std::make_pair(h, mesh.HLength(h))); }
39 }
40 for (Vertex v : verts) if(v != kInvalidId) {
41 for(Halfedge h : mesh.VOutgoings(v)){
42 hqueue.push(std::make_pair(h, mesh.HLength(h)));
43 }
44 }
45 }
46 }

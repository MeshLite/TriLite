371 Face Trimesh::AddFace(const std::array<std::variant<Vertex, Point3>, 3>& triangle)
372 {
373 for(const auto& [key, f_attr] : attributes_[kFace]) f_attr->IncrementSize();
374 Halfedge h_offset = origin_.size();
375 for (const auto& elem : triangle) {
376 Halfedge h = origin_.size();
377 Vertex v = position_.size();
378 if (std::holds_alternative<Vertex>(elem)) {
379 v = std::get<Vertex>(elem);
380 }else{
381 for(const auto& [key, v_attr] : attributes_[kVertex]) v_attr->IncrementSize();
382 position_.push_back(std::get<Point3>(elem));
383 outgoing_.push_back(kInvalidId);
384 }
385 for(const auto& [key, h_attr] : attributes_[kHalfedge]) h_attr->IncrementSize();
386 origin_.push_back(v);
387 coorigin_.push_back(outgoing_[v]);
388 outgoing_[v] = h;
389 }
390 return Face(h_offset);

392 void Trimesh::RemoveFace(Face f)
393 {
394 for(const auto& [key, f_attr] : attributes_[kFace]) { f_attr->ReplaceErase(f); }
395 for(Halfedge h : FaceHalfedges(f) | std::views::reverse){
396 Halfedge g = origin_.size()-1;
397 Vertex v = origin_[h];
398 if(outgoing_[v] == h) outgoing_[v] = coorigin_[h];
399 else for(Halfedge he : OutgoingHalfedges(v)) if(coorigin_[he] == h) { coorigin_[he] = coorigin_[h]; break; }
400 if(h != g){
401 Vertex u = origin_[g];
402 origin_[h] = u;
403 coorigin_[h] = coorigin_[g];
404 Halfedge* p = nullptr;
405 if(outgoing_[u] == g) { p = &outgoing_[u]; }
406 else for(Halfedge he : OutgoingHalfedges(u)) if(coorigin_[he] == g) { p = &coorigin_[he]; break; }
407 *p = h; // replace g with h and ensure the structure remains sorted from greater to lower
408 for(Halfedge nh = coorigin_[h]; coorigin_[h] > h && coorigin_[h] != kInvalidId; nh = coorigin_[h]) {
409 *p = nh;
410 coorigin_[h] = coorigin_[nh];
411 coorigin_[nh] = h;
412 p = &coorigin_[nh];
413 }
414 }
415 origin_.pop_back();
416 coorigin_.pop_back();
417 for(const auto& [key, h_attr] : attributes_[kHalfedge]) { h_attr->ReplaceErase(h); }
418 if(outgoing_[v] == kInvalidId){
419 for(Halfedge he : OutgoingHalfedges(outgoing_.size()-1)){
420 origin_[he] = v;
421 }
422 outgoing_[v] = outgoing_.back(), outgoing_.pop_back();
423 position_[v] = std::move(position_.back()), position_.pop_back();
424 for(const auto& [key, v_attr] : attributes_[kVertex]) { v_attr->ReplaceErase(v); }
425 }
426 }

430 auto it = attributes_[component].find(key);
431 if (it == attributes_[component].end()) { throw std::invalid_argument("Attribute does not exist."); }
432 auto containerPtr = dynamic_cast<ContainerWrapper<std::vector<T>>*>(it->second.get());
433 if (!containerPtr) { throw std::invalid_argument("Attribute exists but type does not match."); }
434 return std::views::all(containerPtr->container_);

437 std::ranges::ref_view<std::vector<T>> Trimesh::CreateAttribute(MeshComponent component, const std::string& key)
438 {
439 if (attributes_[component].find(key) != attributes_[component].end()) { throw std::invalid_argument("Attribute already exists."); }
440 size_t N = component == kHalfedge ? NumHalfedges() : component == kVertex ? NumVertices() : NumFaces();
441 attributes_[component][key] = std::make_unique<ContainerWrapper<std::vector<T>>>(std::vector<T>(N));
442 return GetAttribute<T>(component, key);

61 struct Vector3Hash {
62 std::size_t operator()(const Vector3d& key) const {
63 const std::hash<double> hasher;
64 size_t result = 0;
65 for (int i = 0; i < 3; ++i) {
66 result ^= hasher(key[i]) + 0x9e3779b9 + (result << 6) + (result >> 2);
67 }
68 return result;
69 }
70 };
71 struct Float3Hash {
72 std::size_t operator()(const std::array<float, 3>& key) const {
73 const std::hash<float> hasher;
74 size_t result = 0;
75 for (int i = 0; i < 3; ++i) {
76 result ^= hasher(key[i]) + 0x9e3779b9 + (result << 6) + (result >> 2);
77 }
78 return result;

81 static std::vector<std::variant<V, Vector3d>> ConvertedToTrimeshPoints(
82 const std::vector<Vector3d>& points) {
83 std::vector<std::variant<V, Vector3d>> triangle_points;
84 std::unordered_map<Vector3d, V, Vector3Hash> position_to_vertex;
85 int n_duplicates = 0;
86 for (size_t i = 0; i < points.size(); i += 3) {
87 for (int j = 0; j < 3; j++) {
88 auto [it, inserted] = position_to_vertex.emplace(
89 points[i + j],
90 static_cast<V>(position_to_vertex.size() + n_duplicates));
91 for (int k = 0; k < j; k++) {
92 if (points[i + j] == points[i + k]) {
93 inserted = true;
94 n_duplicates++;
95 break;
96 }
97 }
98 if (inserted) {
99 triangle_points.push_back(points[i + j]);
100 } else {
101 triangle_points.push_back(it->second);
102 }
103 }
104 }
105 return triangle_points;
106 }

116 static Trimesh ReadAsciiSTL(const std::string& filepath) {
117 std::ifstream file(filepath);
118 std::string line, token;
119 std::vector<Vector3d> positions;
120 Vector3d normal, vertex;
121 while (std::getline(file, line)) {
122 std::istringstream linestream(line);
123 linestream >> token;
124 if (token == "facet") {
125 linestream >> token;
126 linestream >> normal[0] >> normal[1] >> normal[2];
127 } else if (token == "vertex") {
128 linestream >> vertex[0] >> vertex[1] >> vertex[2];
129 positions.push_back(vertex);
130 } else if (token == "endfacet") {
131 // Remove vertices if they don't count to three (error in file)
132 positions.resize(positions.size() - (positions.size() % 3));
133 }
134 }
135 positions.resize(positions.size() - (positions.size() % 3));
136 return Trimesh(ConvertedToTrimeshPoints(positions));
137 }
138 static Trimesh ReadBinarySTL(const std::string& filepath) {
139 const std::vector<unsigned char> binaryData = ReadBinaryFile(filepath);
140 int nTriangles = *reinterpret_cast<const int*>(binaryData.data() + 80);
141 std::vector<Vector3d> points(3 * nTriangles);
142 float x;
143 for (int i = 0; i < nTriangles; ++i) {
144 auto ptr = binaryData.data() + 80 + 4 + 50 * i;
145 for (int j : std::views::iota(1, 4)) {
146 for (int k : std::views::iota(0, 3)) {
147 memcpy(&x, ptr + j * 12 + k * 4, 4);
148 points[3 * i + j - 1][k] = x;
149 }
150 }
151 }
152 return Trimesh(ConvertedToTrimeshPoints(points));
153 }

168 static Trimesh ReadOBJ(const std::string& filepath) {
169 std::ifstream file(filepath);
170 std::string line;
171 std::vector<Vector3d> points;
172 std::vector<std::variant<V, Vector3d>> trimesh_points;
173 std::vector<V> id_to_vid;
174 Vector3d p;
175 std::string s_index;
176 V count_vids = 0;
177 while (std::getline(file, line)) {
178 std::istringstream lineStream(line);
179 std::string lineType;
180 lineStream >> lineType;
181 if (lineType == "v") {
182 lineStream >> p.x() >> p.y() >> p.z();
183 points.push_back(std::move(p));
184 id_to_vid.push_back({kInvalidId});
185 } else if (lineType == "f") {
186 std::vector<int> indices;
187 while (lineStream >> s_index) {
188 s_index = s_index.substr(0, s_index.find('/'));
189 indices.push_back(std::stoi(s_index) - 1);
190 }
191 for (size_t i = 1; i < indices.size() - 1; ++i) {
192 int idx[] = {indices[0], indices[i], indices[i + 1]};
193 if (idx[0] == idx[1] || idx[0] == idx[2] || idx[1] == idx[2]) {
194 continue;
195 }
196 for (int j : idx) {
197 if (id_to_vid[j] != kInvalidId) {
198 trimesh_points.push_back(id_to_vid[j]);
199 } else {
200 trimesh_points.push_back(points[j]);
201 id_to_vid[j] = {count_vids++};
202 }
203 }
204 }
205 }
206 }
207
208 return Trimesh(trimesh_points);

250 static Trimesh ReadBinaryPLY(const std::string& filepath) {
251 std::ifstream file(filepath, std::ios::binary);
252 std::string line;
253 bool header_ended = false;
254 int point_count = 0;
255 int face_count = 0;
256 bool is_big_endian = false;
257 while (std::getline(file, line)) {
258 std::istringstream iss(line);
259 std::string token;
260 iss >> token;
261 if (line.find("binary_big_endian") != std::string::npos) {
262 is_big_endian = true;
263 } else if (token == "end_header") {
264 header_ended = true;
265 break;
266 } else if (token == "element") {
267 iss >> token;
268 if (token == "vertex") {
269 iss >> point_count;
270 } else if (token == "face") {
271 iss >> face_count;
272 }
273 }
274 }
275 if (!header_ended) {
276 throw std::runtime_error("PLY end_header is missing.");
277 }
278 auto convert_endian = [](float& value) {
279 char* value_ptr = reinterpret_cast<char*>(&value);
280 std::reverse(value_ptr, value_ptr + sizeof(float));
281 };
282 std::vector<Vector3d> points(point_count);
283 for (int i = 0; i < point_count; ++i) {
284 std::array<float, 3> point;
285 for (float& xyz : point) {
286 file.read(reinterpret_cast<char*>(&xyz), sizeof(float));
287 if (is_big_endian) {
288 convert_endian(xyz);
289 }
290 }
291 points[i] = Vector3d{point[0], point[1], point[2]};
292 }
293 V count_vids = 0;
294 std::vector<V> id_to_vid(point_count, kInvalidId);
295 std::vector<std::variant<V, Vector3d>> trimesh_points;
296 for (int f = 0; f < face_count; ++f) {
297 unsigned char vertex_count;
298 file.read(reinterpret_cast<char*>(&vertex_count), sizeof(vertex_count));
299 std::vector<int> indices(vertex_count);
300 for (int j = 0; j < vertex_count; ++j) {
301 int vertex_index;
302 file.read(reinterpret_cast<char*>(&vertex_index), sizeof(vertex_index));
303 indices[j] = vertex_index;
304 }
305 for (size_t i = 1; i < indices.size() - 1; ++i) {
306 int idx[] = {indices[0], indices[i], indices[i + 1]};
307 if (idx[0] == idx[1] || idx[0] == idx[2] || idx[1] == idx[2]) {
308 continue;
309 }
310 for (int j : idx) {
311 if (id_to_vid[j] != kInvalidId) {
312 trimesh_points.push_back(id_to_vid[j]);
313 } else {
314 trimesh_points.push_back(points[j]);
315 id_to_vid[j] = count_vids++;
316 }
317 }
318 }
319 }
320 return Trimesh(trimesh_points);
321 }

427 static void WriteSTL(const std::string& filepath, const Trimesh& mesh,
428 bool binary_mode) {
429 if (binary_mode) {
430 std::unordered_map<std::array<float, 3>, V, Float3Hash> position_to_id;
431 position_to_id.reserve(mesh.NumVertices());
432 std::ofstream file(filepath, std::ios::binary);
433 char header[80] = {};
434 file.write(header, sizeof(header));
435 uint32_t num_triangles = static_cast<uint32_t>(mesh.NumFaces());
436 file.write(reinterpret_cast<const char*>(&num_triangles),
437 sizeof(num_triangles));
438 auto write_float3 = [&file, &position_to_id](const Vector3d& vec,
439 V v = kInvalidId) {
440 std::array<float, 3> data{static_cast<float>(vec[0]),
441 static_cast<float>(vec[1]),
442 static_cast<float>(vec[2])};
443 if (v != kInvalidId) {
444 // Displace the point until it is unique
445 for (size_t i = 0;
446 position_to_id.find(data) != position_to_id.end() &&
447 position_to_id[data] != v;
448 ++i) {
449 if (!std::isfinite(data[i % 3])) {
450 data[i % 3] = -std::numeric_limits<float>::infinity();
451 }
452 data[i % 3] = std::nextafter(
453 data[i % 3], std::numeric_limits<float>::infinity());
454 }
455 position_to_id[data] = v;
456 }
457 file.write(reinterpret_cast<const char*>(&data), sizeof(data));
458 };
459 for (F f : mesh.Faces()) {
460 write_float3(mesh.FNormal(f));
461 for (V v : mesh.FVertices(f)) {
462 write_float3(mesh.VPosition(v), v);
463 }
464 uint16_t attribute_byte_count = 0;
465 file.write(reinterpret_cast<const char*>(&attribute_byte_count),
466 sizeof(attribute_byte_count));
467 }
468 file.close();
469 } else {
470 double increment = pow(10, 1 - kAsciiDigitsPrecision_);
471 std::vector<Vector3d> positions{
472 std::ranges::to<std::vector>(mesh.Positions())};
473 std::vector<V> ids{std::ranges::to<std::vector>(mesh.Vertices())};
474 std::sort(ids.begin(), ids.end(), [&positions](int u, int v) {
475 return positions[u][0] < positions[v][0];
476 });
477
478 for (size_t i = 1; i < ids.size(); i++) {
479 V u = ids[i - 1], v = ids[i];
480 double ref =
481 std::max(std::nextafter(positions[u][0],
482 std::numeric_limits<double>::infinity()),
483 positions[u][0] + abs(positions[u][0]) * increment);
484 positions[v][0] = std::max(positions[v][0], ref);
485 }
486 std::ofstream file(filepath);
487 file << std::setprecision(kAsciiDigitsPrecision_) << "solid mesh\n";
488 for (F f : mesh.Faces()) {
489 file << " facet normal " << mesh.FNormal(f).transpose() << "\n";
490 file << " outer loop\n";
491 for (V v : mesh.FVertices(f)) {
492 file << " vertex " << positions[v].transpose() << "\n";
493 }
494 file << " endloop\n";
495 file << " endfacet\n";
496 }
497 file << "endsolid mesh\n";
498 file.close();
499 }
500 }
501 static void WritePLY(const std::string& filepath, const Trimesh& mesh,
502 bool binary_mode) {
503 std::ofstream file = binary_mode ? std::ofstream(filepath, std::ios::binary)
504 : std::ofstream(filepath);
505 if (!binary_mode) {
506 file << std::setprecision(kAsciiDigitsPrecision_);
507 }
508 file << "ply\n";
509 file << (binary_mode ? "format binary_little_endian 1.0\n"
510 : "format ascii 1.0\n");
511 file << "element vertex " << mesh.NumVertices() << "\n";
512 file << "property float x\n"
513 << "property float y\n"
514 << "property float z\n";
515 file << "element face " << mesh.NumFaces() << "\n";
516 file << "property list uchar int vertex_index\n";
517 file << "end_header\n";
518 for (const Vector3d& point : mesh.Positions()) {
519 if (binary_mode) {
520 for (float coord : point) {
521 file.write(reinterpret_cast<const char*>(&coord), sizeof(float));
522 }
523 } else {
524 file << point.transpose() << "\n";
525 }
526 }
527 for (F f : mesh.Faces()) {
528 unsigned char num_vertices = 3;
529 if (binary_mode) {
530 file.write(reinterpret_cast<const char*>(&num_vertices),
531 sizeof(num_vertices));
532 for (V v : mesh.FVertices(f)) {
533 file.write(reinterpret_cast<const char*>(&v), sizeof(v));
534 }
535 } else {
536 file << 3 << " " << mesh.HStart(3 * f) << " " << mesh.HStart(3 * f + 1)
537 << " " << mesh.HStart(3 * f + 2) << "\n";
538 }
539 }
540 file.close();

544Trimesh IO::ReadMeshFile(const std::string& filepath) {
545 std::ifstream file(filepath);
546 if (!file.is_open()) {
547 throw std::runtime_error("Could not open file for reading: " + filepath);
548 }
549 file.close();
550 size_t dotPos = filepath.rfind('.');
551 if (dotPos == std::string::npos) {
552 throw std::runtime_error("No file extension found");
553 }
554 std::string extension = filepath.substr(dotPos + 1);
555 std::transform(extension.begin(), extension.end(), extension.begin(),
556 [](unsigned char c) { return std::tolower(c); });
557 if (extension == "stl") {
558 return ReadSTL(filepath);
559 } else if (extension == "obj") {
560 return ReadOBJ(filepath);
561 } else if (extension == "off") {
562 return ReadOFF(filepath);
563 } else if (extension == "ply") {
564 return ReadPLY(filepath);
565 } else {
566 throw std::runtime_error("Unsupported file format: " + extension);
567 }
568}

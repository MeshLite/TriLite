76 static void ClampEdgeLengths(Trimesh& mesh, double min_length,
77 double max_length, int niters = 5) {
78 for (int i = 0; i < niters; i++) {
79 bool stop = true;
80 for (H h : mesh.Halfedges()) {
81 if (mesh.HLength(h) > max_length || mesh.HLength(h) < min_length) {
82 stop = false;
83 break;
84 }
85 }
86 if (stop) {
87 break;
88 }
89 F nf = mesh.NumFaces();
90 std::vector<V> edge_to_v(mesh.NumHalfedges(), kInvalidId);
91 std::vector<F> rm_faces;
92 std::vector<std::vector<H>> edge_halfedges(mesh.NumHalfedges());
93 for (H h : mesh.Halfedges()) {
94 edge_halfedges[h] = std::ranges::to<std::vector>(mesh.EdgeHalfedges(h));
95 }
96 for (F f = 0; f < nf; f++) {
97 int ct = 0;
98 H he{kInvalidId};
99 for (H h : mesh.FHalfedges(f)) {
100 if (edge_to_v[h] != kInvalidId ||
101 mesh.HLength(edge_halfedges[h].front()) > max_length) {
102 ++ct;
103 he = h;
104 }
105 }
106 if (ct > 1) {
107 rm_faces.push_back(f);
108 std::vector<V> verts = {mesh.HStart(3 * f), mesh.HStart(3 * f + 1),
109 mesh.HStart(3 * f + 2)};
110 std::array<std::variant<V, Vector3d>, 3> centroids = {
111 mesh.HCentroid(3 * f), mesh.HCentroid(3 * f + 1),
112 mesh.HCentroid(3 * f + 2)};
113 V num_v = mesh.NumVertices();
114 for (int j : {0, 1, 2}) {
115 if (edge_to_v[3 * f + j] != kInvalidId) {
116 assert(edge_to_v[3 * f + j] < mesh.NumVertices());
117 centroids[j] = edge_to_v[3 * f + j];
118 } else {
119 bool flag = false;
120 for (H g : edge_halfedges[3 * f + j]) {
121 flag |= (g == 3 * f + j);
122 edge_to_v[g] = num_v;
123 }
124 assert(flag);
125 ++num_v;
126 }
127 }
128 mesh.AddFace(centroids);
129 mesh.AddFace({verts[0], edge_to_v[3 * f], edge_to_v[3 * f + 2]});
130 mesh.AddFace({verts[1], edge_to_v[3 * f + 1], edge_to_v[3 * f]});
131 mesh.AddFace({verts[2], edge_to_v[3 * f + 2], edge_to_v[3 * f + 1]});
132 } else if (ct == 1) {
133 rm_faces.push_back(f);
134 std::variant<V, Vector3d> centroid{mesh.HCentroid(he)};
135 if (edge_to_v[he] != kInvalidId) {
136 centroid = edge_to_v[he];
137 } else {
138 for (H g : edge_halfedges[he]) { // mesh.EdgeHalfedges(he)) {
139 edge_to_v[g] = mesh.NumVertices();
140 }
141 }
142 mesh.AddFace(
143 {mesh.HStart(mesh.HPrev(he)), mesh.HStart(he), centroid});
144 centroid = edge_to_v[he];
145 mesh.AddFace({mesh.HEnd(he), mesh.HStart(mesh.HPrev(he)), centroid});
146 }
147 }
148 mesh.RemoveFaces(rm_faces);
149 CollapseSmallEdges(mesh, min_length);
150 }

152 static void CollapseSmallEdges(Trimesh& mesh, double min_length) {
153 for (F f = 0; f < mesh.NumFaces();) {
154 bool flag = true;
155 for (H h : mesh.FHalfedges(f)) {
156 if (mesh.HLength(h) < min_length) {
157 mesh.CollapseEdge(h);
158 flag = false;
159 break;
160 }
161 }
162 f += flag;

165 static std::pair<TL::Vector3d, double> CalculateCircumsphere(
166 const TL::Vector3d& a, const TL::Vector3d& b, const TL::Vector3d& c) {
167 TL::Vector3d ac = c - a;
168 TL::Vector3d ab = b - a;
169 TL::Vector3d abxac = ab.cross(ac);
170
171 TL::Vector3d to_circumsphere_center = (abxac.cross(ab) * ac.squaredNorm() +
172 ac.cross(abxac) * ab.squaredNorm()) /
173 (2.0 * abxac.squaredNorm());
174 double circumsphere_radius = to_circumsphere_center.norm();
175 TL::Vector3d ccs = a + to_circumsphere_center;
176
177 return std::make_pair(ccs, circumsphere_radius);
178 }
179 static bool IsDelaunay(const TL::Trimesh& mesh, TL::H h) {
180 TL::H hopp = mesh.HOpposite(h);
181 if (hopp == TL::kInvalidId) {
182 return true;
183 }
184 TL::Vector3d a = mesh.VPosition(mesh.HStart(h));
185 TL::Vector3d b = mesh.VPosition(mesh.HEnd(h));
186 TL::Vector3d c = mesh.VPosition(mesh.HEnd(mesh.HNext(h)));
187 TL::Vector3d d = mesh.VPosition(mesh.HEnd(mesh.HNext(hopp)));
188 std::pair<TL::Vector3d, double> sphere = CalculateCircumsphere(a, b, c);
189 return (sphere.first - d).norm() >= (sphere.second);
190 }
191 static void MakeDelaunay(TL::Trimesh& mesh) {
192 std::queue<TL::H> edge_queue;
193 for (TL::H h : mesh.Halfedges()) {
194 if (mesh.HOpposite(h) != TL::kInvalidId &&
195 mesh.HOpposite(h) / 3 != mesh.HOpposite(mesh.HNext(h)) / 3) {
196 edge_queue.push(h);
197 }
198 }
199 size_t count = 0;
200 size_t limit = 3 * mesh.NumFaces();
201 while (!edge_queue.empty()) {
202 if (count++ == limit) {
203 break;
204 }
205 TL::H h = edge_queue.front();
206 edge_queue.pop();
207 TL::H hopp = mesh.HOpposite(h);
208 if (hopp != TL::kInvalidId && !IsDelaunay(mesh, h)) {
209 TL::V v1 = mesh.HStart(mesh.HPrev(h));
210 TL::V v2 = mesh.HStart(mesh.HPrev(hopp));
211 bool valid = true;
212 for (TL::H he : mesh.VStartings(v1)) {
213 valid &= (mesh.HEnd(he) != v2);
214 }
215 for (TL::H he : mesh.VStartings(v2)) {
216 valid &= (mesh.HEnd(he) != v1);
217 }
218 if (valid) {
219 mesh.FlipHalfedgeWithOpposite(h);
220 edge_queue.push(mesh.HNext(h));
221 edge_queue.push(mesh.HPrev(h));
222 edge_queue.push(mesh.HNext(hopp));
223 edge_queue.push(mesh.HPrev(hopp));
224 }
225 }
226 }

228 static void RetainLargestComponent(TL::Trimesh& mesh) {
229 std::vector<bool> visited(mesh.NumFaces(), false);
230 std::vector<std::vector<F>> components;
231 for (F f = 0; f < mesh.NumFaces(); ++f) {
232 if (!visited[f]) {
233 std::vector<F> component;
234 std::vector<F> stack;
235 stack.push_back(f);
236 while (!stack.empty()) {
237 F current = stack.back();
238 stack.pop_back();
239 if (visited[current]) continue;
240 visited[current] = true;
241 component.push_back(current);
242 for (F neighbor : mesh.FNeighbors(current)) {
243 if (!visited[neighbor]) {
244 stack.push_back(neighbor);
245 }
246 }
247 }
248 components.push_back(component);
249 }
250 }
251 size_t max_id = 0;
252 for (size_t i = 1; i < components.size(); i++) {
253 if (components[i].size() > components[max_id].size()) {
254 max_id = i;
255 }
256 }
257 std::vector<F> faces_to_delete;
258 for (size_t i = 0; i < components.size(); i++) {
259 if (i != max_id) {
260 for (F f : components[i]) {
261 faces_to_delete.push_back(f);
262 }
263 }
264 }
265 mesh.RemoveFaces(faces_to_delete);
266 }

281 static BVHNode* ConstructBVH(
282 std::vector<std::pair<TL::F, std::array<Vector3d, 3>>>& triangles,
283 size_t start, size_t end, int depth = 0) {
284 BVHNode* node = new BVHNode();
285 assert(end > start);
286 if (end - start == 1) {
287 node->triangle_ = triangles[start];
288 for (const auto& vertex : triangles[start].second) {
289 node->bbox_.extend(vertex);
290 }
291 } else {
292 for (size_t i = start; i < end; ++i) {
293 for (const auto& vertex : triangles[i].second) {
294 node->bbox_.extend(vertex);
295 }
296 }
297 Vector3d extents = node->bbox_.sizes();
298 int axis = 0;
299 if (extents[1] > extents[0]) axis = 1;
300 if (extents[2] > extents[axis]) axis = 2;
301 std::sort(triangles.begin() + start, triangles.begin() + end,
302 [axis](const std::pair<TL::F, std::array<Vector3d, 3>>& a,
303 const std::pair<TL::F, std::array<Vector3d, 3>>& b) {
304 return a.second[0][axis] < b.second[0][axis];
305 });
306
307 size_t mid = start + (end - start) / 2;
308 node->left_ = ConstructBVH(triangles, start, mid, depth + 1);
309 node->right_ = ConstructBVH(triangles, mid, end, depth + 1);
310 }
311 return node;
312 }

314 static bool DoesIntersect(
315 BVHNode* node, const std::pair<TL::F, std::array<Vector3d, 3>>& tri) {
316 if (!node || !node->bbox_.intersects(Eigen::AlignedBox3d(tri.second[0])
317 .extend(tri.second[1])
318 .extend(tri.second[2]))) {
319 return false;
320 }
321 if (node->triangle_) {
322 const auto& node_tri = node->triangle_.value();
323 if (tri.first != node_tri.first &&
324 Intersection::TriIntersectTri2(tri.second, node_tri.second)) {
325 return true;
326 }
327 }
328 return DoesIntersect(node->left_, tri) || DoesIntersect(node->right_, tri);
329 }

331 // Main function to find all auto-intersections in a mesh
332 static std::vector<TL::F> FindSelfIntersections(TL::Trimesh& mesh,
333 double shrink_factor = 1e-8) {
334 if (mesh.NumFaces() == 0) {
335 return {};
336 }
337 std::vector<std::pair<TL::F, std::array<Vector3d, 3>>> triangles;
338 for (TL::F f : mesh.Faces()) {
339 if (mesh.FArea(f) > 1e-10) {
340 Vector3d centroid = mesh.FCentroid(f);
341 std::array<Vector3d, 3> tri;
342 for (auto [i, position] : std::views::enumerate(mesh.FPositions(f))) {
343 tri[i] = position + (centroid - position) * shrink_factor;
344 }
345 triangles.emplace_back(f, tri);
346 }
347 }
348 std::vector<TL::F> self_intersection_faces;
349 BVHNode* bvh_root = ConstructBVH(triangles, 0, triangles.size());
350 for (const auto& tri : triangles) {
351 if (DoesIntersect(bvh_root, tri)) {
352 self_intersection_faces.push_back(tri.first);
353 }
354 }
355 delete bvh_root;
356 return self_intersection_faces;
357 }

374 // Function to check if a point is inside a triangle using barycentric

377 static bool SegmentIntersectsTriangle(const std::array<Vector3d, 2>& seg,
378 const std::array<Vector3d, 3>& tri) {
379 Vector3d N = (tri[1] - tri[0]).cross(tri[2] - tri[0]).normalized();
380 Vector3d ray_vector = (seg[1] - seg[0]).normalized();
381 Vector3d edge1 = tri[1] - tri[0];
382 Vector3d edge2 = tri[2] - tri[0];
383 Vector3d ray_cross_e2 = ray_vector.cross(edge2);
384 double det = edge1.dot(ray_cross_e2);
385 if (abs(det) < 1e-11) {
386 // Edge is coplanar with the triangle
387 if (std::abs((tri[0] - seg[0]).dot(N)) < 1e-11) {
388 return CoplanarSegmentIntersectsTriangle(seg, tri);
389 }
390 } else {
391 // Edge is not coplanar with the triangle
392 return NoCoplanarSegmentIntersectsTriangle(seg, tri);
393 }
394 return false;
395 }
396 static bool NoCoplanarSegmentIntersectsTriangle(
397 const std::array<Vector3d, 2>& seg,
398 const std::array<Vector3d, 3>& tri) {
399 Vector3d ray_origin = seg[0];
400 Vector3d ray_vector = (seg[1] - seg[0]).normalized();
401 Vector3d edge1 = tri[1] - tri[0];
402 Vector3d edge2 = tri[2] - tri[0];
403 Vector3d ray_cross_e2 = ray_vector.cross(edge2);
404 double det = edge1.dot(ray_cross_e2);
405 assert(abs(det) > 1e-12);
406 double inv_det = 1.0 / det;
407 Vector3d s = ray_origin - tri[0];
408 double u = inv_det * s.dot(ray_cross_e2);
409 if (u < 0.0 || u > 1.0) {
410 return false;
411 }
412 Vector3d s_cross_e1 = s.cross(edge1);
413 double v = inv_det * ray_vector.dot(s_cross_e1);
414 if (v < 0.0 || u + v > 1.0) {
415 return false;
416 }
417 double t = inv_det * edge2.dot(s_cross_e1);
418 if (t >= 0.0 && t <= (seg[1] - seg[0]).norm()) {
419 return true;
420 } else {
421 return false;
422 }
423 }
424 static bool EdgeEdgeTest(const Vector3d& v0, const Vector3d& u0,
425 const Vector3d& u1, short i0, short i1, double ax,
426 double ay) {
427 double bx = u0[i0] - u1[i0];
428 double by = u0[i1] - u1[i1];
429 double cx = v0[i0] - u0[i0];
430 double cy = v0[i1] - u0[i1];
431 double f = ay * bx - ax * by;
432 double d = by * cx - bx * cy;
433 if ((f > 0 && d >= 0 && d <= f) || (f < 0 && d <= 0 && d >= f)) {
434 double e = ax * cy - ay * cx;
435 if (f > 0) {
436 if (e >= 0 && e <= f) return true;
437 } else {
438 if (e <= 0 && e >= f) return true;
439 }
440 }
441 return false;
442 }

444 static bool EdgeAgainstTriEdges(const Vector3d& v0, const Vector3d& v1,
445 const Vector3d& u0, const Vector3d& u1,
446 const Vector3d& u2, short i0, short i1) {
447 double ax = v1[i0] - v0[i0];
448 double ay = v1[i1] - v0[i1];
449 if (EdgeEdgeTest(v0, u0, u1, i0, i1, ax, ay)) {
450 return true;
451 }
452 if (EdgeEdgeTest(v0, u1, u2, i0, i1, ax, ay)) {
453 return true;
454 }
455 if (EdgeEdgeTest(v0, u2, u0, i0, i1, ax, ay)) {
456 return true;
457 }
458 return false;
459 }

461 static bool CoplanarSegmentIntersectsTriangle(
462 const std::array<Vector3d, 2>& seg,
463 const std::array<Vector3d, 3>& tri) {
464 short i0, i1;
465 Vector3d edge = seg[1] - seg[0];
466 Vector3d n = (tri[1] - tri[0]).cross(tri[2] - tri[0]);
467 Vector3d a = n.cwiseAbs();
468 if (a[0] > a[1]) {
469 if (a[0] > a[2]) {
470 i0 = 1;
471 i1 = 2;
472 } else {
473 i0 = 0;
474 i1 = 1;
475 }
476 } else {
477 if (a[1] > a[2]) {
478 i0 = 0;
479 i1 = 2;
480 } else {
481 i0 = 0;
482 i1 = 1;
483 }
484 }
485 if (EdgeEdgeTest(seg[0], tri[0], tri[1], i0, i1, edge[i0], edge[i1])) {
486 return true;
487 }
488 if (EdgeEdgeTest(seg[0], tri[1], tri[2], i0, i1, edge[i0], edge[i1])) {
489 return true;
490 }
491 if (EdgeEdgeTest(seg[0], tri[2], tri[0], i0, i1, edge[i0], edge[i1])) {
492 return true;
493 }
494 if (PointInTri(seg[0], tri[0], tri[1], tri[2], i0, i1)) {
495 return true;
496 }
497 return false;
498 }
499
500 static bool PointInTri(const Vector3d& p, const Vector3d& u0,
501 const Vector3d& u1, const Vector3d& u2, short i0,
502 short i1) {
503 double a, b, c, d0, d1, d2;
504 a = u1[i1] - u0[i1];
505 b = -(u1[i0] - u0[i0]);
506 c = -a * u0[i0] - b * u0[i1];
507 d0 = a * p[i0] + b * p[i1] + c;
508
509 a = u2[i1] - u1[i1];
510 b = -(u2[i0] - u1[i0]);
511 c = -a * u1[i0] - b * u1[i1];
512 d1 = a * p[i0] + b * p[i1] + c;
513
514 a = u0[i1] - u2[i1];
515 b = -(u0[i0] - u2[i0]);
516 c = -a * u2[i0] - b * u2[i1];
517 d2 = a * p[i0] + b * p[i1] + c;
518
519 if (d0 * d1 > 0.0 && d0 * d2 > 0.0) {
520 return true;
521 }
522 return false;
523 }

527void Processing::DecimateMesh(Trimesh& mesh, size_t target_face_count) {
528 std::vector<double> edge_lengths(mesh.NumHalfedges());
529 auto compare = [&edge_lengths](H h, H g) {
530 return edge_lengths[h] < edge_lengths[g] ||
531 (edge_lengths[h] == edge_lengths[g] && h < g);
532 };
533 std::set<H, decltype(compare)> hset(compare);
534 for (H h : mesh.Halfedges()) {
535 edge_lengths[h] = mesh.HLength(h);
536 assert(std::isfinite(edge_lengths[h]));
537 hset.insert(h);
538 }
539 while (mesh.NumFaces() > target_face_count && !hset.empty()) {
540 auto minh = *hset.begin();
541 hset.erase(minh);
542 if (minh >= mesh.NumHalfedges()) {
543 continue;
544 }
545 std::array<V, 2> verts = {mesh.HStart(minh), mesh.HEnd(minh)};
546 V last_vert_id = mesh.NumVertices() - 1;
547 auto [rm_faces, rm_vertices] = mesh.CollapseEdge(minh);
548 for (const V& rem_v : rm_vertices) {
549 for (V& v : verts) {
550 if (v == rem_v) {
551 v = kInvalidId;
552 } else if (v == last_vert_id) {
553 v = rem_v;
554 }
555 }
556 last_vert_id--;
557 }
558 for (F f : rm_faces) {
559 for (H h : {3 * f, 3 * f + 1, 3 * f + 2}) {
560 if (h < mesh.NumHalfedges()) {
561 hset.erase(h);
562 }
563 }
564 }
565 for (V v : verts) {
566 if (v != kInvalidId) {
567 for (H h : mesh.VStartings(v)) {
568 hset.erase(h);
569 }
570 }
571 }
572 for (F f : rm_faces) {
573 for (H h : {3 * f, 3 * f + 1, 3 * f + 2}) {
574 if (h < mesh.NumHalfedges()) {
575 edge_lengths[h] = mesh.HLength(h);
576 hset.insert(h);
577 }
578 }
579 }
580 for (V v : verts) {
581 if (v != kInvalidId) {
582 for (H h : mesh.VStartings(v)) {
583 edge_lengths[h] = mesh.HLength(h);
584 hset.insert(h);
585 }
586 }
587 }
588 }

591void Processing::FillMeshHoles(Trimesh& mesh, size_t target_hole_count) {
592 std::unordered_set<H> boundary_edges;
593 std::vector<std::vector<H>> polygons;
594 for (H st_h : mesh.BoundaryHalfedges()) {
595 if (!boundary_edges.count(st_h)) {
596 polygons.push_back(
597 std::ranges::to<std::vector>(mesh.HHalfedgesAroundHole(st_h)));
598 boundary_edges.insert(polygons.back().begin(), polygons.back().end());
599 }
600 }
601 std::sort(polygons.begin(), polygons.end(),
602 [](auto& A, auto& B) { return A.size() < B.size(); });
603 for (const auto& [id, polygon] : std::views::enumerate(polygons)) {
604 if (polygons.size() - id == target_hole_count) {
605 break;
606 }
607 assert(polygon.size() >= 3);
608 std::vector<double> sum_angle(polygon.size());
609 auto set_sum_angle = [&mesh, &polygon, &sum_angle](size_t i) {
610 H h = polygon[i];
611 sum_angle[i] = 0;
612 for (H he : mesh.HConnectionsAroundStart(h)) {
613 sum_angle[i] += std::acos(
614 std::clamp(mesh.HGeometry(he).normalized().dot(
615 -mesh.HGeometry(mesh.HPrev(he)).normalized()),
616 -1.0, 1.0));
617 }
618 };
619 auto cmp = [&mesh, &sum_angle](std::pair<size_t, double> a,
620 std::pair<size_t, double> b) {
621 return a.second < b.second;
622 };
623 std::priority_queue<std::pair<size_t, double>,
624 std::vector<std::pair<size_t, double>>, decltype(cmp)>
625 q(cmp);
626 for (size_t i = 0; i < polygon.size(); i++) {
627 set_sum_angle(i);
628 q.push(std::make_pair(i, sum_angle[i]));
629 }
630
631 std::set<std::pair<size_t, H>> s;
632 for (size_t i = 0; i < polygon.size(); i++) {
633 s.insert(std::make_pair(i, polygon[i]));
634 }
635 while (!q.empty() && s.size() >= 3) {
636 auto [i1, angle] = q.top();
637 q.pop();
638 if (angle != sum_angle[i1]) {
639 continue;
640 }
641 H h1 = polygon[i1];
642 auto it = s.find(std::make_pair(i1, h1));
643 if (it == s.end()) {
644 continue;
645 }
646 auto [i0, h0] = *((it == s.begin()) ? std::prev(s.end()) : std::prev(it));
647 auto [i2, h2] = *((std::next(it) == s.end()) ? s.begin() : std::next(it));
648
649 mesh.AddFace({mesh.HStart(h0), mesh.HStart(h2), mesh.HStart(h1)});
650 sum_angle[i0] += std::acos(std::clamp(
651 mesh.HGeometry(mesh.NumHalfedges() - 3)
652 .normalized()
653 .dot(-mesh.HGeometry(mesh.NumHalfedges() - 1).normalized()),
654 -1.0, 1.0));
655 q.push(std::make_pair(i0, sum_angle[i0]));
656 sum_angle[i2] += std::acos(std::clamp(
657 mesh.HGeometry(mesh.NumHalfedges() - 2)
658 .normalized()
659 .dot(-mesh.HGeometry(mesh.NumHalfedges() - 3).normalized()),
660 -1.0, 1.0));
661 q.push(std::make_pair(i2, sum_angle[i2]));
662 s.erase(it);
663 }
664 }

667void Processing::TaubinSmoothing(Trimesh& mesh, int iterations, double lambda,
668 double mu) {
669 std::vector<Vector3d> new_positions(mesh.NumVertices());
670
671 for (int iter = 0; iter < iterations; ++iter) {
672 // Laplacian smoothing step (positive weight)
673 for (V v : mesh.Vertices()) {
674 Vector3d laplacian(0, 0, 0);
675 size_t valence = 0;
676 for (H h : mesh.VStartings(v)) {
677 laplacian += mesh.VPosition(mesh.HEnd(h));
678 ++valence;
679 }
680 laplacian /= static_cast<double>(valence);
681 new_positions[v] =
682 mesh.VPosition(v) + lambda * (laplacian - mesh.VPosition(v));
683 }
684 for (V v : mesh.Vertices()) {
685 mesh.VPosition(v) = new_positions[v];
686 }
687
688 // Laplacian smoothing step (negative weight, inverse smoothing)
689 for (V v : mesh.Vertices()) {
690 Vector3d laplacian(0, 0, 0);
691 size_t valence = 0;
692 for (H h : mesh.VStartings(v)) {
693 laplacian += mesh.VPosition(mesh.HEnd(h));
694 ++valence;
695 }
696 laplacian /= static_cast<double>(valence);
697 new_positions[v] =
698 mesh.VPosition(v) + mu * (laplacian - mesh.VPosition(v));
699 }
700 for (V v : mesh.Vertices()) {
701 mesh.VPosition(v) = new_positions[v];
702 }
703 }
704}

714void Processing::PrintabilityHeuristics(Trimesh& mesh, int niters) {
715 if (mesh.NumFaces() == 0) {
716 return;
717 }
718 RetainLargestComponent(mesh);
719 std::vector<double> lengths(mesh.NumHalfedges());
720 for (H h : mesh.Halfedges()) {
721 lengths[h] = mesh.HLength(h);
722 }
723 std::sort(lengths.begin(), lengths.end());
724 double max_length = 2.0 * lengths[lengths.size() / 2];
725 double min_length = 0.5 * lengths[lengths.size() / 2];
726 for (int i = 0; i < niters; i++) {
727 ClampEdgeLengths(mesh, min_length, max_length);
728 TL::Processing::RemoveSelfIntersections(mesh);
729 mesh.DisconnectFacesUntilManifold();
730 RetainLargestComponent(mesh);
731 TL::Processing::FillMeshHoles(mesh, 0);
732 mesh.DisconnectFacesUntilManifold();
733 RetainLargestComponent(mesh);
734 MakeDelaunay(mesh);
735 TL::Processing::TaubinSmoothing(mesh);
736 }
737}

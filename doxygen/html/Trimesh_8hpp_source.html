508 class ContainerWrapper : public BaseContainerWrapper {
509 public:
510 Container container_;
511 explicit ContainerWrapper(const Container& container)
512 : container_(container) {}

514 return std::make_unique<ContainerWrapper<Container>>(*this);

517 container_[i] = std::move(container_.back());

598H Trimesh::HOpposite(H h) const {
599 for (H nh : VStartings(HStart(HNext(h)))) {
600 if (HEnd(nh) == HStart(h) && HStart(nh) == HEnd(h)) {
601 return nh;
602 }
603 }
604 return kInvalidId;

620std::generator<H> Trimesh::HConnectionsAroundStart(H st_h) const {
621 H h = st_h;
622 while (HOpposite(h) != kInvalidId) {
623 assert(EdgeIsManifold(h));
624 h = HPrevAroundStart(h);
625 if (h == st_h) {
626 break;
627 }
628 }
629 st_h = h;
630 do {
631 assert(EdgeIsManifold(h));
632 co_yield h;
633 h = HNextAroundStart(h);
634 } while (h != st_h && h != kInvalidId);
635}
636std::generator<H> Trimesh::HHalfedgesAroundHole(H st_h) const {
637 assert(HOpposite(st_h) == kInvalidId);
638 H h = st_h;
639 do {
640 co_yield h;
641 h = HNext(h);
642 while (HOpposite(h) != kInvalidId) {
643 assert(EdgeIsManifold(h));
644 h = HPrevAroundStart(h);
645 }
646 } while (h != st_h);
647}

711inline std::pair<Vector3d, Vector3d> Trimesh::FBoundingBox(F f) const {
712 return {VPosition(HStart(3 * f))
713 .cwiseMin(VPosition(HStart(3 * f + 1)))
714 .cwiseMin(VPosition(HStart(3 * f + 2))),
715 VPosition(HStart(3 * f))

720 return (VPosition(HStart(3 * f)) + VPosition(HStart(3 * f + 1)) +

724std::generator<H> Trimesh::EdgeHalfedges(H h) const {
725 for (H he : VStartings(HStart(h))) {
726 if (HEnd(he) == HEnd(h)) {
727 co_yield (he);
728 }
729 }
730 for (H he : VStartings(HEnd(h))) {
731 if (HEnd(he) == HStart(h)) {
732 co_yield (he);
733 }
734 }
735}
736auto Trimesh::EdgeFaces(H h) const {
737 return std::views::transform(EdgeHalfedges(h),
738 [this](H h) { return HFace(h); });
739}
740bool Trimesh::EdgeIsManifold(H h) const {
741 for (H g : EdgeHalfedges(h)) {
742 if (g != h && g != HOpposite(h)) {
743 return false;
744 }
745 }
746 return true;
747}
748inline auto Trimesh::BoundaryHalfedges() const {
749 return std::views::filter(Halfedges(),

752std::pair<Vector3d, Vector3d> Trimesh::BoundingBox() const {
753 if (position_.empty()) {
754 throw std::runtime_error("Mesh has no vertices.");
755 }
756 Vector3d min = position_.front();
757 Vector3d max = min;
758 for (const auto& pos : position_) {
759 min = min.cwiseMin(pos);
760 max = max.cwiseMax(pos);
761 }
762 return {min, max};
763}
764Vector3d Trimesh::Centroid() const {
765 if (position_.empty()) {
766 throw std::runtime_error("Mesh has no vertices.");
767 }
768 Vector3d centroid = Vector3d::Zero();
769 for (const auto& pos : position_) {
770 centroid += pos;
771 }
772 centroid /= static_cast<double>(position_.size());
773 return centroid;
774}
775F Trimesh::AddFace(const std::array<std::variant<V, Vector3d>, 3>& triangle) {
776 for (int i = 0; i < 3; i++) {
777 if (std::holds_alternative<V>(triangle[i])) {
778 for (int j = 0; j < i; j++) {
779 if (triangle[i] == triangle[j]) {
780 throw std::invalid_argument(
781 "AddFace called with a degenerated triangle parameter");
782 }
783 }
784 }
785 }
786 for (const auto& [key, f_attr] : attributes_[kFace]) {
787 f_attr->IncrementSize();
788 }
789 H h_offset = hStart_.size();
790 for (const auto& elem : triangle) {
791 H h = hStart_.size();
792 V v = position_.size();
793 if (std::holds_alternative<V>(elem)) {
794 v = std::get<V>(elem);
795 } else {
796 for (const auto& [key, v_attr] : attributes_[kVertex]) {
797 v_attr->IncrementSize();
798 }
799 position_.push_back(std::get<Vector3d>(elem));
800 vStart_.push_back(kInvalidId);
801 }
802 for (const auto& [key, h_attr] : attributes_[kHalfedge]) {
803 h_attr->IncrementSize();
804 }
805 hStart_.push_back(v);
806 hCoStart_.push_back(vStart_.at(v));
807 vStart_.at(v) = h;
808 }
809 return HFace(h_offset);
810}
811std::vector<V> Trimesh::RemoveFace(F f) {
812 std::vector<V> removed_vertices;
813 for (const auto& [key, f_attr] : attributes_[kFace]) {
814 f_attr->ReplaceErase(f);
815 }
816 for (H h : std::views::reverse(FHalfedges(f))) {
817 H g = hStart_.size() - 1;
818 V v = hStart_[h];
819 if (vStart_[v] == h) {
820 vStart_[v] = hCoStart_[h];
821 } else {
822 for (H he : VStartings(v)) {
823 if (hCoStart_[he] == h) {
824 hCoStart_[he] = hCoStart_[h];
825 break;
826 }
827 }
828 }
829 if (h != g) {
830 V u = hStart_[g];
831 hStart_[h] = u;
832 hCoStart_[h] = hCoStart_[g];
833 H* p = nullptr;
834 if (vStart_[u] == g) {
835 p = &vStart_[u];
836 } else {
837 for (H he : VStartings(u)) {
838 if (hCoStart_[he] == g) {
839 p = &hCoStart_[he];
840 break;
841 }
842 }
843 }
844 *p = h; // replace g with h and ensure the structure remains sorted
845 // from greater to lower
846 for (H nh = hCoStart_[h]; hCoStart_[h] > h && hCoStart_[h] != kInvalidId;
847 nh = hCoStart_[h]) {
848 *p = nh;
849 hCoStart_[h] = hCoStart_[nh];
850 hCoStart_[nh] = h;
851 p = &hCoStart_[nh];
852 }
853 }
854 hStart_.pop_back();
855 hCoStart_.pop_back();
856 for (const auto& [key, h_attr] : attributes_[kHalfedge]) {
857 h_attr->ReplaceErase(h);
858 }
859 if (vStart_[v] == kInvalidId) {
860 removed_vertices.push_back(v);
861 for (H he : VStartings(vStart_.size() - 1)) {
862 hStart_[he] = v;
863 }
864 vStart_[v] = vStart_.back(), vStart_.pop_back();
865 position_[v] = std::move(position_.back()), position_.pop_back();
866 for (const auto& [key, v_attr] : attributes_[kVertex]) {
867 v_attr->ReplaceErase(v);
868 }
869 }
870 }
871 return removed_vertices;
872}
873std::vector<V> Trimesh::RemoveFaces(std::vector<F> faces) {
874 std::ranges::sort(faces);
875 const auto [first, last] = std::ranges::unique(faces);
876 faces.erase(first, last);
877 std::vector<V> removed_vertices;
878 for (std::size_t i = 0; i < faces.size();) {
879 F f = faces[i];
880 std::vector<V> removed = RemoveFace(f);
881 removed_vertices.insert(removed_vertices.end(), removed.begin(),
882 removed.end());
883 if (faces.back() == NumFaces()) {
884 faces.pop_back();
885 } else {
886 i++;
887 }
888 }
889 return removed_vertices;
890}
891std::pair<std::vector<F>, std::vector<V>> Trimesh::CollapseEdge(H h) {
892 std::vector<V> removed_vertices;
893 std::vector<F> removed_faces = EdgeFaces(h) | std::ranges::to<std::vector>();
894 std::array<V, 2> verts = {HStart(h), HEnd(h)};
895 Vector3d midpoint = (VPosition(verts[0]) + VPosition(verts[1])) / 2.0;
896 V last_vert_id = NumVertices() - 1;
897 for (size_t i = 0; i < removed_faces.size(); i++) {
898 std::vector<V> rm_verts = RemoveFace(removed_faces[i]);
899 removed_vertices.insert(removed_vertices.end(), rm_verts.begin(),
900 rm_verts.end());
901 for (size_t j = i + 1; j < removed_faces.size(); j++) {
902 if (removed_faces[j] == NumFaces()) {
903 removed_faces[j] = removed_faces[i];
904 }
905 }
906 }
907 for (const V& rem_v : removed_vertices) {
908 for (V& v : verts) {
909 if (v == rem_v) {
910 v = kInvalidId;
911 } else if (v == last_vert_id) {
912 v = rem_v;
913 }
914 }
915 last_vert_id--;
916 }
917 if (verts[0] == kInvalidId) {
918 std::swap(verts[0], verts[1]);
919 }
920 if (verts[0] == kInvalidId) {
921 return std::make_pair(std::move(removed_faces),
922 std::move(removed_vertices));
923 }
924 position_[verts[0]] = midpoint;
925 std::vector<H> v_starts;
926 for (V vert : verts) {
927 if (vert != kInvalidId) {
928 for (H he : VStartings(vert)) {
929 v_starts.push_back(he);
930 }
931 }
932 }
933 std::sort(v_starts.begin(), v_starts.end());
934 vStart_[verts[0]] = kInvalidId;
935 for (H he : v_starts) {
936 hStart_[he] = verts[0];
937 hCoStart_[he] = vStart_[verts[0]];
938 vStart_[verts[0]] = he;
939 }
940 if (verts[1] != kInvalidId) {
941 removed_vertices.push_back(verts[1]);
942 if (verts[1] != vStart_.size() - 1) {
943 for (H he : VStartings(vStart_.size() - 1)) {
944 hStart_[he] = verts[1];
945 }
946 vStart_[verts[1]] = vStart_.back();
947 position_[verts[1]] = std::move(position_.back());
948 }
949 vStart_.pop_back();
950 position_.pop_back();
951 for (const auto& [key, v_attr] : attributes_[kVertex]) {
952 v_attr->ReplaceErase(verts[1]);
953 }
954 }
955 return std::make_pair(std::move(removed_faces), std::move(removed_vertices));
956}
957void Trimesh::SplitEdge(H h) {
958 std::variant<V, Vector3d> new_p = HCentroid(h);
959 std::array<std::variant<V, Vector3d>, 3> tri;
960 for (H he : std::ranges::to<std::vector>(EdgeHalfedges(h))) {
961 tri = {new_p, HEnd(he), HStart(HPrev(he))};
962 std::rotate(tri.begin(), tri.begin() + (3 - (he % 3)) % 3, tri.end());
963 AddFace(tri);
964 new_p = NumVertices() - 1;
965 tri = {HStart(he), new_p, HStart(HPrev(he))};
966 std::rotate(tri.begin(), tri.begin() + (3 - (he % 3)) % 3, tri.end());
967 AddFace(tri);
968 RemoveFace(HFace(he));
969 }
970}
971void Trimesh::FlipHalfedgeWithOpposite(H h) {
972 H hopp = HOpposite(h);
973 if (hopp == kInvalidId) {
974 throw std::invalid_argument("h param needs opposite to perform edge flip");
975 }
976 std::array<V, 4> verts{HStart(h), HEnd(h), HEnd(HNext(h)), HEnd(HNext(hopp))};
977 if (verts[2] == verts[3]) {
978 throw std::invalid_argument("Opp faces are identical, edge flip fails");
979 }
980 std::array<std::variant<V, Vector3d>, 3> tri = {verts[3], verts[2], verts[0]};
981 std::rotate(tri.begin(), tri.begin() + (3 - (h % 3)) % 3, tri.end());
982 AddFace(tri);
983 RemoveFace(HFace(h));
984 tri = {verts[2], verts[3], verts[1]};
985 std::rotate(tri.begin(), tri.begin() + (3 - (hopp % 3)) % 3, tri.end());
986 AddFace(tri);
987 RemoveFace(HFace(hopp));
988}
989void Trimesh::DisconnectFace(F f) {
990 AddFace({VPosition(HStart(3 * f)), VPosition(HStart(3 * f + 1)),
991 VPosition(HStart(3 * f + 2))});
992 RemoveFace(f);
993}
994void Trimesh::DisconnectFacesUntilManifoldEdges() {
995 for (F f : std::views::reverse(Faces())) {
996 for (H h : FHalfedges(f)) {
997 if (!EdgeIsManifold(h)) {
998 DisconnectFace(f);
999 break;
1000 }
1001 }
1002 }
1003}
1004void Trimesh::DisconnectFacesUntilManifoldVertices() {
1005 std::queue<V> q;
1006 for (V v : Vertices()) {
1007 if (!VIsManifold(v)) {
1008 q.push(v);
1009 }
1010 }
1011 while (!q.empty()) {
1012 V v = q.front();
1013 q.pop();
1014 if (VIsManifold(v)) {
1015 continue;
1016 }
1017 std::unordered_set<V> s;
1018 for (H h : HConnectionsAroundStart(VStarting(v))) {
1019 s.insert(h);
1020 }
1021 for (H h : VStartings(v)) {
1022 if (!s.count(h)) {
1023 F f{HFace(h)};
1024 for (V u : FVertices(f)) {
1025 q.push(u);
1026 }
1027 DisconnectFace(f);
1028 }
1029 }
1030 }
1031}
1032void Trimesh::DisconnectFacesUntilManifold() {
1033 DisconnectFacesUntilManifoldEdges();
1034 DisconnectFacesUntilManifoldVertices();
1035}

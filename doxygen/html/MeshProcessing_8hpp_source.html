41void DecimateMesh(Trimesh& mesh, size_t target_face_count) {
42 std::vector<double> edge_lengths(mesh.NumHalfedges());
43 auto compare = [&edge_lengths](H h, H g) {
44 return edge_lengths[h] < edge_lengths[g] ||
45 (edge_lengths[h] == edge_lengths[g] && h < g);
46 };
47 std::set<H, decltype(compare)> hset(compare);
48 for (H h : mesh.Halfedges()) {
49 edge_lengths[h] = mesh.HLength(h);
50 hset.insert(h);
51 }
52 while (mesh.NumFaces() > target_face_count && !hset.empty()) {
53 auto minh = *hset.begin();
54 hset.erase(minh);
55 if (minh >= mesh.NumHalfedges()) {
56 continue;
57 }
58 std::array<V, 2> verts = {mesh.HStart(minh), mesh.HEnd(minh)};
59 V last_vert_id = mesh.NumVertices() - 1;
60 auto [rm_faces, rm_vertices] = mesh.EdgeCollapse(minh);
61 for (const V& rem_v : rm_vertices) {
62 for (V& v : verts) {
63 if (v == rem_v) {
64 v = kInvalidId;
65 } else if (v == last_vert_id) {
66 v = rem_v;
67 }
68 }
69 last_vert_id--;
70 }
71 for (F f : rm_faces) {
72 for (H h : {3 * f, 3 * f + 1, 3 * f + 2}) {
73 if (h < mesh.NumHalfedges()) {
74 hset.erase(h);
75 }
76 }
77 }
78 for (V v : verts) {
79 if (v != kInvalidId) {
80 for (H h : mesh.VStartings(v)) {
81 hset.erase(h);
82 }
83 }
84 }
85 for (F f : rm_faces) {
86 for (H h : {3 * f, 3 * f + 1, 3 * f + 2}) {
87 if (h < mesh.NumHalfedges()) {
88 edge_lengths[h] = mesh.HLength(h);
89 hset.insert(h);
90 }
91 }
92 }
93 for (V v : verts) {
94 if (v != kInvalidId) {
95 for (H h : mesh.VStartings(v)) {
96 edge_lengths[h] = mesh.HLength(h);
97 hset.insert(h);
98 }
99 }
100 }
101 }
102}

104void FillMeshHoles(Trimesh& mesh, size_t target_hole_count) {
105 std::unordered_set<H> boundary_edges;
106 std::vector<std::vector<H>> polygons;
107 for (H st_h : mesh.BoundaryHalfedges()) {
108 if (!boundary_edges.count(st_h)) {
109 polygons.push_back(
110 std::ranges::to<std::vector>(mesh.HHalfedgesSharingHole(st_h)));
111 boundary_edges.insert(polygons.back().begin(), polygons.back().end());
112 }
113 }
114 std::sort(polygons.begin(), polygons.end(),
115 [](auto& A, auto& B) { return A.size() < B.size(); });
116 for (const auto& [id, polygon] : std::views::enumerate(polygons)) {
117 if (polygons.size() - id == target_hole_count) {
118 break;
119 }
120 assert(polygon.size() >= 3);
121 V v0 = mesh.HStart(polygon[0]);
122 for (size_t i = 1; i < polygon.size() - 1; ++i) {
123 V v1 = mesh.HStart(polygon[i]);
124 V v2 = mesh.HStart(polygon[i + 1]);
125 mesh.AddFace({v0, v2, v1});
126 }
127 }
128}

130void TaubinSmoothing(Trimesh& mesh, int iterations, double lambda, double mu) {
131 std::vector<Eigen::Vector3d> new_positions(mesh.NumVertices());
132
133 for (int iter = 0; iter < iterations; ++iter) {
134 // Laplacian smoothing step (positive weight)
135 for (V v : mesh.Vertices()) {
136 Eigen::Vector3d laplacian(0, 0, 0);
137 size_t valence = 0;
138 for (H h : mesh.VStartings(v)) {
139 laplacian += mesh.VPosition(mesh.HEnd(h));
140 ++valence;
141 }
142 laplacian /= static_cast<double>(valence);
143 new_positions[v] =
144 mesh.VPosition(v) + lambda * (laplacian - mesh.VPosition(v));
145 }
146 for (V v : mesh.Vertices()) {
147 mesh.VPosition(v) = new_positions[v];
148 }
149
150 // Laplacian smoothing step (negative weight, inverse smoothing)
151 for (V v : mesh.Vertices()) {
152 Eigen::Vector3d laplacian(0, 0, 0);
153 size_t valence = 0;
154 for (H h : mesh.VStartings(v)) {
155 laplacian += mesh.VPosition(mesh.HEnd(h));
156 ++valence;
157 }
158 laplacian /= static_cast<double>(valence);
159 new_positions[v] =
160 mesh.VPosition(v) + mu * (laplacian - mesh.VPosition(v));
161 }
162 for (V v : mesh.Vertices()) {
163 mesh.VPosition(v) = new_positions[v];
164 }
165 }
166}
